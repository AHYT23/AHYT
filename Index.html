const initApp = () => { const { db, collection, addDoc, query, orderBy, onSnapshot, serverTimestamp } = window; if (!db) { setTimeout(initApp, 500); return; } // PUBLIER window.publishPost = async function() { const catElem = document.getElementById('categorie'); // Assorti à ton ID sans accent const textElem = document.getElementById('postContent'); if (!catElem || !textElem || textElem.value.trim().length < 2) return; try { await addDoc(collection(db, "posts"), { category: catElem.value, content: textElem.value, timestamp: serverTimestamp() }); textElem.value = ""; alert("✅ Message envoyé !"); } catch (e) { alert("Erreur : " + e.message); } }; // CLASSER PAR CATÉGORIES const q = query(collection(db, "posts"), orderBy("timestamp", "desc")); onSnapshot(q, (snapshot) => { const feed = document.getElementById('feed'); feed.innerHTML = ""; const categories = ["Vie de Campus", "Cours", "GBAIRAI"]; const sections = {}; categories.forEach(cat => { const div = document.createElement('div'); div.style = "margin-bottom: 25px; display: none;"; div.innerHTML = `

${cat}

`; sections[cat] = div; feed.appendChild(div); }); snapshot.forEach((doc) => { const data = doc.data(); const postDiv = document.createElement('div'); postDiv.className = 'post'; // Utilise ton style.css postDiv.innerHTML = `
${data.category}
${data.content}
`; if (sections[data.category]) { sections[data.category].appendChild(postDiv); sections[data.category].style.display = "block"; } }); }); }; initApp();
